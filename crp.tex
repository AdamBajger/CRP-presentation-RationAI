\section[Motivation]{Motivation}
% ═══════════════════════════════════════════════════════════════════════════

\begin{frame}{Motivation}
ViTs win vision benchmarks. We want post-hoc explanations that are:
\begin{itemize}
  \item faithful (conserve the prediction),
  \item spatial (pixel-level heat-maps),
  \item conceptual (per-head, per-channel, per-token).
\end{itemize}
Gradient saliency loses conservation through softmax, bilinear matmul, residual adds, LayerScale.

AttnLRP~\cite{achtibat2024attnlrp} — transformer-aware LRP rules.
CRP~\cite{achtibat2023crp} — lifts attribution to per-concept maps.

Our work: ground both into a single \texttt{zennit}-compatible composite for DINOv3~\cite{darcet2024dinov3} and standard timm ViTs~\cite{dosovitskiy2021vit}.
\end{frame}

% ═══════════════════════════════════════════════════════════════════════════
\section[LRP]{LRP}
% ═══════════════════════════════════════════════════════════════════════════

\begin{frame}{LRP-0 and $\eps$-LRP}
\begin{block}{LRP-0 (Bach et al.\ 2015)}
For $y_j = \sum_i w_{ij}\, x_i + b_j$:
\[
  \Rel_i = \sum_j \frac{w_{ij}\, x_i}{\sum_k w_{kj}\, x_k}\, \Rel_j
\]
Exact conservation on linear layers.
\end{block}
\begin{block}{$\eps$-LRP (stabilised)}
\[
  \Rel_i = \sum_j \frac{w_{ij}\, x_i}{\eps\cdot \mathrm{sign}(z_j) + z_j}\, \Rel_j
\]
$\eps$ kills small-denominator amplification. Conservation $\sum \Rel_i \approx \sum \Rel_j$ within $\eps$.
\end{block}

\structure{Problem}: softmax, bilinear matmul, residual adds, LayerScale are NOT linear.
Stock LRP cannot route relevance through them.
\end{frame}

\begin{frame}{AttnLRP}
Per-non-linearity rules that preserve LRP-0 conservation~\cite{achtibat2024attnlrp}:

\begin{itemize}
  \item Identity rule for normalisations (Eq.~9) — softmax, LayerNorm, GELU.
  \item Bilinear rule (Prop.~3.3) for $Q@K^\top$ and $\mathrm{attn}@V$.
  \item Stop-gradient on std inside LayerNorm.
\end{itemize}

\begin{block}{Our additions}
\begin{itemize}
  \item AlphaBeta on bilinear — Bach~\cite{bach2015lrp} generalised; better magnitude control.
  \item Residual ratio (Otsuki split) and LayerScale uniform rule.
  \item Attention unfolding: substitute attention with named-submodule form so concepts can hook on $Q$, $K$, $V$, head, channel, token positions.
\end{itemize}
\end{block}
\end{frame}

\begin{frame}{Attention unfolding}
Stock attention forward is one expression:
\[
  \mathrm{Attn}(\mathbf{x}) \;=\; \mathrm{softmax}\!\left(\tfrac{Q K^\top}{\sqrt{d_h}}\right) V .
\]
Per-tensor LRP rules need a handle on each non-linear op ($Q K^\top$,
softmax, the $\mathrm{attn} \cdot V$ bilinear). We rewrite the same
forward as a composition of named ops so that per-tensor rules can
attach (and any concept can address $Q$, $K$, $V$ slices). Forward
output is bit-identical; the rewrite changes only the autograd graph.

\begin{block}{Hookable 3D sites (one shape contract $\reals^{B\times N\times D}$)}
\centering $q,\ k,\ v$ (after qkv-split, before per-head reshape) and the post-projection output.
\end{block}
\end{frame}

\section[Rules]{LRP rules}
% ═══════════════════════════════════════════════════════════════════════════

\begin{frame}{Rules}
\small
\begin{tabular}{lll}
\toprule
Module & Rule & Source \\
\midrule
\texttt{Linear}, \texttt{Conv2d}      & $\eps$-LRP                          & \cite{bach2015lrp}; \texttt{zennit.Epsilon}~\cite{anders2021zennit} \\
\texttt{LayerNorm}                    & Stop-grad on std + identity         & AttnLRP \S 3.2.2~\cite{achtibat2024attnlrp} \\
\texttt{GELU}                         & Identity (Eq.~9)                    & \cite{achtibat2024attnlrp} \\
\texttt{softmax}                      & Identity                            & \cite{achtibat2024attnlrp} \\
\texttt{scale\_q} (constant mul)      & Identity                            & graph constant \\
\texttt{qk\_scores}, \texttt{context} & AlphaBeta on bilinear               & \cite{bach2015lrp} + our derivation \\
Residual additions                    & Otsuki ratio split                   & magnitude-weighted \\
LayerScale                            & Uniform $\half$ allocation           & AttnLRP Eq.~7~\cite{achtibat2024attnlrp} \\
\bottomrule
\end{tabular}

\vspace{4mm}
Vanilla forward, LRP only on backward.
Forwards are bit-identical; only the autograd graph changes during attribution.
\end{frame}

\begin{frame}{$\eps$-LRP}
For $z_j = \sum_i w_{ij}\, x_i + b_j$ in any \texttt{Linear}:
\[
  \Rel_i = \sum_j \frac{w_{ij}\, x_i}{\eps\cdot\mathrm{sign}(z_j) + z_j}\, \Rel_j
\]

\begin{itemize}
  \item Workhorse — every \texttt{Linear} / \texttt{Conv2d} (\texttt{qkv}, \texttt{proj}, \texttt{fc1}, \texttt{fc2}, \ldots).
  \item Conservation exact modulo $\eps$.
\end{itemize}
\end{frame}

\begin{frame}{Identity rule}
For an element-wise nonlinearity $y_i = f(x_i)$, LRP-0 reduces to
\[
  \Rel_i \;=\; \frac{y_i}{\eps\cdot\mathrm{sign}(y_i) + y_i}\, \Rel^{\mathrm{out}}_i
  \;\approx\; \Rel^{\mathrm{out}}_i \qquad (|y_i| \gg \eps).
\]

\begin{itemize}
  \item \texttt{softmax}: relevance passes straight through — does NOT route via Jacobian.
  \item \texttt{LayerNorm}: \texttt{stop\_gradient}(std) $\to$ identity on affine.
  \item \texttt{GELU}, \texttt{scale\_q} (constant multiplier): same form.
\end{itemize}

Conservation: $\sum \Rel_i = \sum \Rel^{\mathrm{out}}_i$.
\end{frame}

\begin{frame}{Bilinear rule: AlphaBeta}
For $Y = A\,B$, decompose by sign: $A^\pm = \max/\min(A, 0)$, $B^\pm = \max/\min(B, 0)$.
\[
  Y^+ = A^+B^+ + A^-B^- \geq 0, \qquad
  Y^- = A^+B^- + A^-B^+ \leq 0
\]
Define $F = \Rel_Y/(Y^+ + \eps)$, $G = \Rel_Y/(Y^- - \eps)$. Then
\[
  \Rel_A = \half \Bigl\{ A^+ \odot \bigl[\alpha\,(FB^{+\top}) + \beta\,(GB^{-\top})\bigr]
                       + A^- \odot \bigl[\alpha\,(FB^{-\top}) + \beta\,(GB^{+\top})\bigr] \Bigr\}
\]
(symmetric for $\Rel_B$).
\medskip
\textit{Note: $\odot$ denotes the Hadamard product (element-wise multiplication).}
\vspace{2mm}
\begin{itemize}
  \item Conservation: $\sum \Rel_A + \sum \Rel_B = (\alpha+\beta)\sum \Rel_Y$; exact when $\alpha+\beta=1$.
  \item Default $\alpha=0.5, \beta=0.5$ — balanced, tightest magnitude control.
  \item AttnLRP's $2Y+\eps$ rule amplifies by $\sim 1/\eps$ at near-orthogonal $Q,K$.
        Sign-splitting denominators avoid this structurally.
\end{itemize}
\end{frame}

\begin{frame}{Residual rule: Otsuki ratio split}
For $y = x + \mathrm{branch}$:
\[
  \Rel_x = \Rel_y \odot \frac{|x|}{|x| + |\mathrm{branch}| + \eps},
  \qquad
  \Rel_{\mathrm{branch}} = \Rel_y \odot \frac{|\mathrm{branch}|}{|x| + |\mathrm{branch}| + \eps}
\]
\medskip
\textit{Note: $\odot$ denotes the Hadamard product (element-wise multiplication).}
\vspace{2mm}
\begin{itemize}
  \item Magnitude-weighted: bigger-signal branch gets more relevance.
  \item Bare autograd: $\Rel_x = \Rel_{\mathrm{branch}} = \Rel_y$ — sums to $2\Rel_y$
        per block $\Rightarrow$ $2^{24}\!\times$ blow-up on a 24-block ViT-L.
  \item Conservation: $\sum \Rel_x + \sum \Rel_{\mathrm{branch}} \approx \sum \Rel_y$.
\end{itemize}
\end{frame}

\begin{frame}{LayerScale: uniform $\half$ allocation}
LayerScale~\cite{touvron2021cait}: $\mathrm{branch} \to \gamma \odot \mathrm{branch}$ with $\gamma \in \reals^D$ init $\sim 10^{-4}$.

\begin{block}{Why bare autograd fails}
$\nabla_{\mathrm{branch}} = \nabla_y \cdot \gamma$ with $\gamma \sim 10^{-4}$, 24 blocks $\times$ 2 LayerScales
$\Rightarrow$ $\Rel$ deflates by $10^{-200}$ — numerical death.
\end{block}

\begin{block}{Fix (AttnLRP Eq.~7)}
Treat $\gamma$ as a leaf parameter absorbing half the relevance:
\[
  \Rel_{\mathrm{branch}} = \half \Rel_y,\qquad \Rel_\gamma = \half \Rel_y\ \mbox{(absorbed)}.
\]
Replaces ``$\times\gamma$'' with constant ``$\times\half$'' per LayerScale.
\end{block}
\end{frame}

\begin{frame}{Conservation}
\centering\small
\begin{tabular}{lll}
\toprule
Rule & Conservation & Comment \\
\midrule
$\eps$-LRP on Linear     & $\sum \Rel_i = \sum \Rel_j$ mod $\eps$       & exact \\
Identity (sm / LN / GELU) & $\sum \Rel_i = \sum \Rel_j$ exact            & passes through \\
Bilinear AlphaBeta       & $= (\alpha+\beta)\sum \Rel_Y$               & exact when $\alpha+\beta=1$ \\
Residual ratio           & $\sum \Rel_x + \sum \Rel_b \approx \sum \Rel_y$ & magnitude-weighted \\
LayerScale uniform       & $\Rel_{\mathrm{branch}} = \half \Rel_y$       & $\gamma$ absorbs the rest \\
\bottomrule
\end{tabular}

\vspace{4mm}
Bare autograd on 24-block Eva DINOv3 inflates $\sum\!\Rel$ by $\sim\!10^{21}$.
Composite-corrected: $\sum\!\Rel$ within $\sim\!10^0$–$10^1$ of target logit.
Audit metric: $\sum\!\Rel\,/\,\mathrm{logit}$ per block.
\end{frame}

\section[Concepts]{Concepts}
% ═══════════════════════════════════════════════════════════════════════════

\begin{frame}{Concepts}
\begin{block}{LRP $\to$ one heatmap per input. CRP $\to$ one heatmap per concept.}
A concept partitions a hidden tensor into named ``parts''. To
attribute concept $c$ at layer $\ell$: mask all parts $\neq c$, continue
backward as usual.
\end{block}

\begin{block}{Three concepts, four hookable sites — all share one shape}
Every site is $(B, N, D)$ — the 3D tensor at the attention boundary.
Any concept hooks at any site; meaning shifts with the site.
\end{block}

\begin{itemize}
  \item HeadConcept — partitions $D$ into $H$ contiguous head slices ($D/H$ wide).
  \item EmbeddingDimConcept — one part per single embedding dim ($D$ parts total).
  \item TokenConcept — one part per token position; optional
        \texttt{token\_filter=slice(...)} restricts to cls / register / spatial.
\end{itemize}
\end{frame}

\begin{frame}{HeadConcept}
For relevance $\Rel \in \reals^{B \times N \times D}$ at any 3D site,
$\Rel$ reshaped to $(B, N, H, d_h)$ has heads on a separate axis.

\begin{block}{Mask for head $h$}
\[
  \Rel'_{b, n, h', d} = \begin{cases} \Rel_{b, n, h', d} & h' = h \\ 0 & \mbox{otherwise} \end{cases}
\]
\end{block}

\begin{block}{Per-head attribution}
\[
  a_{b, h} = \sum_{n \in \mathit{token\_filter}}\;\; \sum_{d=0}^{d_h-1} \Rel_{b, n, h, d}
\]
\end{block}

Question: ``which head fired on which input regions?''
\end{frame}

\begin{frame}{EmbeddingDimConcept}
Finer than HeadConcept: one part per single dim of $D$.

\begin{block}{Mask for dim set $S \subseteq \{0,\ldots,D-1\}$}
\[
  \Rel'_{b, n, d} = \begin{cases} \Rel_{b, n, d} & d \in S \\ 0 & \mbox{otherwise} \end{cases}
\]
\end{block}

\begin{block}{Per-dim attribution}
\[
  a_{b, d} = \sum_{n \in \mathit{token\_filter}} \Rel_{b, n, d}
\]
\end{block}

Question: ``which single embedding direction carries the prediction at this layer?''
\end{frame}

\begin{frame}{TokenConcept}
Partitions on the token axis. For prefix-vs-spatial separation, use
\texttt{token\_filter} (e.g.\ \texttt{slice(0, 5)} = cls + 4 register tokens
on DINOv3; \texttt{slice(5, None)} = spatial patches only).

\begin{block}{Mask for token set $T \subseteq [0, N)$}
\[
  \Rel'_{b, n, d} = \begin{cases} \Rel_{b, n, d} & n \in T \\ 0 & \mbox{otherwise} \end{cases}
\]
\end{block}

\begin{block}{Per-position attribution}
\[
  a_{b, n} = \sum_{d=0}^{D-1} \Rel_{b, n, d}, \qquad n \in \mathit{token\_filter}
\]
\end{block}

Question: ``did this cls / register / patch position drive the prediction?''\\
DINOv3 register tokens~\cite{darcet2024dinov3}: 4 ``signal-sinks'' separate from cls.
\end{frame}

\section[Findings]{Key findings}
% ═══════════════════════════════════════════════════════════════════════════

\begin{frame}{Key findings so far}
\begin{itemize}
  \item Bare AttnLRP fails on DINOv3. 24 EvaBlocks with LayerScale
        $\sim\!10^{-4}$ + the $2Y+\eps$ bilinear rule overflow fp32 by
        mid-depth. No single literature remedy fixes it; the working
        composite needs three rules simultaneously: AlphaBeta
        bilinear, Otsuki ratio residual, uniform LayerScale.
  \item AlphaBeta-on-bilinear replaces $2Y+\eps$. Sign-splitting
        denominators bound per-element amplification structurally —
        $\sim$ 19 OOM smaller $\max|R|$ vs the standard rule on
        DINOv3 ViT-L, while preserving spatial focus.
  \item Unfolded attention enables CRP. Stock attention has no
        named submodules. Substituting in a forward-equivalent unfolded
        form exposes $Q$, $K$, $V$, scores, softmax, output as concept hooks.
  \item Single shape contract. All four 3D hookable sites carry
        $(B, N, D)$; any of the three concepts attaches at any site.
        Three classes replaced our earlier six-class taxonomy.
  \item Residual flow on FunnyBirds vit\_base: 70--82\% of
        relevance flows through the skip path mid-stack — motivates the
        move to smaller (12-block) backbones.
\end{itemize}
\end{frame}

\begin{frame}{Trained \texttt{vit\_small} on three datasets}
\centering\small
\begin{tabular}{lllp{0.32\textwidth}}
\toprule
Dataset & train-val acc & test acc & Interpretation \\
\midrule
dSprites      & $0.999$ & $0.999$ & 3-class shape task — ceiling. \\
FunnyBirds    & $0.972$ & $0.984$ & 50-class part-based; OneCycle in 25 epochs. \\
ColoredMNIST  & $0.978$ & $0.127$ & intentionally biased — colour shortcut. \\
\bottomrule
\end{tabular}

\vspace{4mm}
\begin{itemize}
  \item ColoredMNIST is the deliverable bias probe — known failure
        mode (colour, not shape). XAI heatmaps should light up
        colour-bearing pixels, not stroke geometry; otherwise the
        method is mis-identifying the cause of the prediction.
  \item The other two are sanity-checks: known parts (FunnyBirds),
        known shape pixels (dSprites). Heatmaps should align with the
        ground truth.
\end{itemize}
\end{frame}

\begin{frame}{Cross-check XAI baselines}
Both baselines hook the post-softmax attention weight tensor
$A \in \reals^{B \times H \times N \times N}$ and compute $\partial y / \partial A$.

\begin{block}{LeGrad~\cite{bousselham2024legrad}}
\[
  s^{(\ell)} \;=\; \mathrm{mean}_h\!\bigl[(\nabla\!_{\!A^{(\ell)}}\, y\;\odot\; A^{(\ell)})_+\bigr]_{\,\mathrm{cls},\, \mathrm{patches}}
  \qquad
  \mathrm{heatmap} \;=\; \sum_{\ell \in \mathit{target}} s^{(\ell)}
\]
\end{block}

\begin{block}{Chefer's rollout~\cite{chefer2021transformer}}
\[
  R^{(\ell)} \;=\; \mathrm{mean}_h\!\bigl[(\nabla\!_{\!A^{(\ell)}}\, y\;\odot\; A^{(\ell)})_+\bigr],
  \qquad
  R \;\leftarrow\; R + R^{(\ell)} R \quad (R_0 = I)
\]
\end{block}

Same atomic ingredient; the only difference is the aggregation —
LeGrad averages, Chefer accumulates via matmul rollup.
\end{frame}

\section[Motivation]{Motivation}
% ═══════════════════════════════════════════════════════════════════════════

\begin{frame}{Motivation}{Why CRP for vision transformers?}
\begin{itemize}
  \item ViTs dominate vision benchmarks; we want \alert{post-hoc explanations} that are
    \begin{itemize}
      \item \structure{faithful} (relevance flow conserves the prediction),
      \item \structure{spatial} (heat-maps over input pixels),
      \item \structure{conceptual} (per-channel, per-head, per-token granularity).
    \end{itemize}
  \item Standard saliency maps (gradient-based) lose conservation through softmax,
        bilinear matmul, residual adds and LayerScale.
  \item \alert{AttnLRP} (Achtibat et al.\ 2024) gives transformer-aware LRP rules.
        \alert{CRP} (Achtibat et al.\ 2023) lifts attribution to per-concept maps.
  \item Our work: ground both into a single \texttt{zennit}-compatible composite
        for DINOv3 ViT-L (Eva-style) and standard timm ViTs.
\end{itemize}
\begin{exampleblock}{Talk plan}
Two attention-block diagrams $\to$ rules $\to$ canonizers $\to$ composite $\to$ concepts.\\
The walkthrough notebook follows after these slides.
\end{exampleblock}
\end{frame}

\section[LRP recap]{LRP recap}
% ═══════════════════════════════════════════════════════════════════════════

\begin{frame}{LRP-0 and \texorpdfstring{$\eps$}{epsilon}-LRP}{}
\begin{block}{LRP-0 (Bach et al.\ 2015)}
For $y_j = \sum_i w_{ij}\, x_i + b_j$:
\[
  \Rel_i = \sum_j \frac{w_{ij}\, x_i}{\sum_k w_{kj}\, x_k}\, \Rel_j
  \quad\mbox{(linear layers, exact conservation)}
\]
\end{block}
\begin{block}{$\eps$-LRP (stabilised)}
\[
  \Rel_i = \sum_j \frac{w_{ij}\, x_i}{\eps\cdot \mathrm{sign}(z_j) + z_j}\, \Rel_j,
  \qquad z_j = \sum_i w_{ij}\, x_i
\]
$\eps$ kills small-denominator amplification; conservation $\sum \Rel_i \approx \sum \Rel_j$ within $\eps$.
\end{block}

\begin{itemize}
  \item Implemented in \texttt{zennit} as \texttt{Epsilon} on \texttt{nn.Linear}/\texttt{nn.Conv2d}.
  \item \alert{Problem}: ViT's \alert{softmax}, \alert{bilinear matmul}, \alert{residual adds}, and \alert{LayerScale} are NOT
        linear layers $\Rightarrow$ stock LRP cannot route relevance through them.
\end{itemize}
\end{frame}

\begin{frame}{AttnLRP (Achtibat et al.\ 2024)}{The recipe}
Per-non-linearity rules that preserve the LRP-0 conservation property.

\begin{itemize}
  \item \alert{Identity rule} for normalisations (Eq.~9): pass relevance through softmax/LN/GELU
  \item \alert{Bilinear rule} for $Q@K^\top$ and $\mathrm{attn}@V$ (Prop.~3.3): $2y+\eps$ denominator
  \item \alert{Stop-gradient on std} inside LayerNorm (treats normalisation factor as constant)
\end{itemize}

\begin{block}{What we add (\texttt{zennit-crp})}
\begin{itemize}
  \item \alert{Bilinear AlphaBeta rule}: Bach 2015 generalised to bilinear (sign-decomposed),
        better magnitude control on near-orthogonal $Q,K$.
  \item \alert{Residual ratio rule} (Otsuki split) and \alert{LayerScale uniform rule} (Eq.~7) for Eva blocks.
  \item \alert{Attention unfolding}: substitute Eva attention with one with named submodules,
        so \alert{concepts} can hook on $Q$, $K$, $V$, head, channel, register-token positions.
\end{itemize}
\end{block}
\end{frame}

\bottomrule
\end{tabular}

\vspace{4mm}
\structure{Vanilla forward, LRP only on backward.} The atomic modules
(\texttt{BilinearMatmul}, \texttt{SoftmaxAlongLastDim}, \texttt{ScaleByConstant}, \ldots)
have plain PyTorch \texttt{forward}s and live inside trainable heads.
At attribution time the canonizer rebinds each forward to route through an
autograd \texttt{Function} whose \texttt{backward} implements the LRP rule.
On context-exit the original forward is restored.
\end{frame}

\begin{frame}{$\eps$-LRP for linears (\texttt{zennit.Epsilon})}{The workhorse}
For $z_j = \sum_i w_{ij}\, x_i + b_j$ inside an \texttt{nn.Linear}:
\[
  \Rel_i = \sum_j \frac{w_{ij}\, x_i}{\eps\cdot\mathrm{sign}(z_j) + z_j}\, \Rel_j
\]

\begin{itemize}
  \item Applied to all \texttt{nn.Linear} and \texttt{nn.Conv2d} via the layer\_map.
  \item In the unfolded Eva attention: \texttt{qkv}, \texttt{proj}, \texttt{q\_norm} (if affine),
        \texttt{k\_norm} (if affine), \texttt{mlp.fc1}, \texttt{mlp.fc2}.
  \item Conservation: $\sum \Rel_i = \sum \Rel_j$ exactly modulo $\eps$.
  \item $\gamma$-LRP variant available via \texttt{AttnLRPGammaComposite} (biases relevance toward positive contributions).
\end{itemize}
\end{frame}

\begin{frame}{Identity rule for normalisations (AttnLRP Eq.~9)}{Softmax, GELU, LayerNorm, ScaleByConstant}
For an element-wise non-linearity $y_i = f(x_i)$, the LRP-0 derivation reduces to:
\[
  \Rel_i = \frac{y_i}{\eps\cdot\mathrm{sign}(y_i) + y_i}\, \Rel^{\mathrm{out}}_i
  \;\approx\; \Rel^{\mathrm{out}}_i
\]
when $f$ contributed to the output ($|y_i|\gg\eps$).

\begin{block}{Where it fires}
\begin{itemize}
  \item \texttt{SoftmaxAlongLastDim} (the \texttt{softmax} node in Diagram~B):
        $\Rel^{\mathrm{scores}} = \Rel^{\mathrm{attn}}$ — softmax does NOT route relevance through its Jacobian.
  \item \texttt{nn.LayerNorm} (\texttt{norm1}, \texttt{norm2}): forward is rebound to use
        \texttt{stop\_gradient(std)}, then identity rule on the affine output.
  \item \texttt{nn.GELU}: same $f(y)/y$ stabilised ratio.
  \item \texttt{ScaleByConstant} (\texttt{scale\_q}, $d_h^{-1/2}$): $\Rel_x = \Rel_y$ — constant has no upstream input.
\end{itemize}
\end{block}

Conservation: $\sum \Rel_i = \sum \Rel^{\mathrm{out}}_i$ exact; relevance flows freely through the normalisation.
\end{frame}

\begin{frame}{Bilinear rule: AlphaBeta on bilinear}{Bach 2015 generalised; the project's single bilinear rule}
For $Y = A\,B$ — the \texttt{qk\_scores} and \texttt{context} matmuls — sign-decompose operands:
$A^\pm = \max/\min(A, 0)$, similarly $B^\pm$. Then
\[
  Y^+ = A^+B^+ + A^-B^- \;(\geq 0), \qquad
  Y^- = A^+B^- + A^-B^+ \;(\leq 0)
\]
Define stabilised ratios $F = \Rel_Y/(Y^+ + \eps)$, $G = \Rel_Y/(Y^- - \eps)$. Then
\[
  \Rel_A = \half \Bigl\{ A^+ \odot \bigl[\alpha\,(FB^{+\top}) + \beta\,(GB^{-\top})\bigr]
                       + A^- \odot \bigl[\alpha\,(FB^{-\top}) + \beta\,(GB^{+\top})\bigr] \Bigr\}
\]
(symmetric for $\Rel_B$).

\begin{itemize}
  \item \alert{Conservation}: $\sum \Rel_A + \sum \Rel_B = (\alpha+\beta)\,\sum \Rel_Y$;
        with $\alpha+\beta = 1$, exact modulo $\eps$.
  \item \alert{Why not AttnLRP Prop.\ 3.3 ($2y+\eps$)?} On near-orthogonal $Q,K$ with $|2Y|\sim\eps$
        in many positions, the factor-2 rule's \texttt{LRP-0 Case~B} cancellation amplifies relevance
        by $\sim 10^{20}$. Sign-splitting denominators avoid this structurally — so we ship only
        AlphaBeta and have deleted the factor-2 variant.
  \item \alert{Default}: $\alpha=0.5,\ \beta=0.5$ — balanced, tightest magnitude control.
\end{itemize}

\begin{exampleblock}{In code}
\texttt{BilinearMatmulAlphaBetaCanonizer} rebinds \texttt{BilinearMatmul.forward} to route
through \texttt{\_AlphaBetaMatmulFn.apply(\ldots)}. Fires on both \texttt{qk\_scores} and
\texttt{context} of every unfolded block (Eva and timm).
\end{exampleblock}
\end{frame}

\begin{frame}{Residual rule: Otsuki ratio split}{For \texttt{add1} and \texttt{add2} in Diagram~B}
For $y = x + \mathrm{branch}$:
\[
  \Rel_x = \Rel_y \odot \frac{|x|}{|x| + |\mathrm{branch}| + \eps},
  \qquad
  \Rel_{\mathrm{branch}} = \Rel_y \odot \frac{|\mathrm{branch}|}{|x| + |\mathrm{branch}| + \eps}
\]

\begin{itemize}
  \item Ratio rule \alert{matches the magnitudes} carried by each branch — the side that holds more
        signal also receives proportionally more relevance.
  \item Bare autograd would give $\Rel_x = \Rel_{\mathrm{branch}} = \Rel_y$ (each branch sees the FULL gradient),
        which sums to $2\Rel_y$ on \alert{every} block $\Rightarrow$ relevance grows $2^{24}$ fold over a 24-block ViT-L.
  \item Conservation: $\sum \Rel_x + \sum \Rel_{\mathrm{branch}} \approx \sum \Rel_y$ exactly modulo $\eps$.
\end{itemize}

\begin{exampleblock}{In code}
The \texttt{EvaBlock} forward is rebound by \texttt{EvaBlockResidualCanonizer}; both block-level adds (\texttt{add1}, \texttt{add2}) route through \texttt{\_ResidualRatioFn}. \texttt{ResidualRatioCanonizer} is the explicit canonizer for the standalone \texttt{ResidualAdd} module.
\end{exampleblock}
\end{frame}

\begin{frame}{LayerScale rule: uniform allocation (AttnLRP Eq.~7)}{For \texttt{ls1} and \texttt{ls2} in Diagram~B}
LayerScale (Touvron et al.\ 2021) multiplies each branch by a learnable per-channel
$\gamma\in\reals^D$ initialised tiny ($\sim 10^{-4}$).

\begin{block}{Why it matters for LRP}
Bare autograd: $\nabla_{\mathrm{branch}} = \nabla_y\cdot\gamma$.\\
With $\gamma\sim 10^{-4}$ and 24 blocks $\times$ 2 LayerScales/block:
$\Rel$ deflates by $10^{-200}$ — numerical death.
\end{block}

\begin{block}{The fix}
Treat $\gamma$ as a leaf parameter that absorbs half the relevance:
\[
  \Rel_{\mathrm{branch}} = \half \Rel_y, \qquad \Rel_\gamma = \half \Rel_y\ \mbox{(absorbed)}
\]
\end{block}

\begin{itemize}
  \item Replaces ``$\times\gamma$'' dampening with constant ``$\times\half$'' per LayerScale.
  \item In code: \texttt{LayerScaleUniformCanonizer} on the standalone \texttt{LayerScaleMul} module,
        or routed inline by \texttt{EvaBlockResidualCanonizer} when \texttt{layerscale\_uniform=True}.
\end{itemize}
\end{frame}

\begin{frame}{Conservation properties — at a glance}
\centering\small
\begin{tabular}{lll}
\toprule
Rule & Conservation & Comment \\
\midrule
$\eps$-LRP on Linear            & $\sum \Rel_i = \sum \Rel_j$ mod $\eps$ & exact \\
Identity (softmax / LN / GELU)  & $\sum \Rel_i = \sum \Rel_j$ exact      & $\Rel$ passes through \\
ScaleByConstant identity        & $\Rel_x = \Rel_y$ exact                 & constant absorbs none \\
Bilinear AlphaBeta              & $= (\alpha+\beta)\sum \Rel_Y$         & exact when $\alpha+\beta=1$ \\
Residual ratio                  & $\sum \Rel_x + \sum \Rel_b \approx \sum \Rel_y$ & magnitude-weighted \\
LayerScale uniform              & $\Rel_{\mathrm{branch}} = \half \Rel_y$ & $\gamma$ absorbs the rest \\
\bottomrule
\end{tabular}

\vspace{4mm}
\begin{itemize}
  \item Multiplied across 24 blocks: bare autograd through Eva yields $\Rel$ inflated by $\sim\!10^{21}$
        relative to the target logit. Composite-corrected: $\sim 10^{0}$ to $10^{1}$.
  \item Ratio of $\sum \Rel$ to the target logit is the per-block conservation \alert{audit metric}.
\end{itemize}
\end{frame}

\section[Canonizers]{Canonizers — taxonomy}
% ═══════════════════════════════════════════════════════════════════════════

\begin{frame}{Two flavours of canonizer}{What \alert{apply} and \alert{remove} actually do}
\begin{block}{Type 1 — \alert{Forward rebind} (\texttt{AttributeCanonizer})}
On \texttt{apply(root)}: walks \texttt{root.named\_modules()}; for each match,
\texttt{setattr(module, "forward", patched\_fn)}. Forward output is bit-identical to vanilla
(only the autograd graph changes). On \texttt{remove()}: \texttt{delattr} restores the bound class method.
\end{block}

\begin{block}{Type 2 — \alert{Module substitution} (custom Canonizer)}
On \texttt{apply(root)}: walks \texttt{root.named\_modules()}; for each \texttt{EvaAttention} found,
constructs an \texttt{EvaAttentionUnfolded} wrapper around it (sharing parameter references), then
\texttt{setattr(parent, "attn", unfolded)}. On \texttt{remove()}: re-binds the original. Used \alert{only} by
\texttt{EvaAttentionSubstitutionCanonizer} — needed because an unfolded attention has \alert{different submodule structure} from the stock \texttt{EvaAttention}, not just a different forward.
\end{block}

\centering\scriptsize
Both types are reversible — composite \texttt{context()} guarantees state restoration on exit.
\end{frame}

\begin{frame}{Canonizer inventory}{What each one does}
\centering\scriptsize
\begin{tabular}{lll}
\toprule
Canonizer & Target & Effect \\
\midrule
\multicolumn{3}{l}{\textbf{In \texttt{crp/transformer\_patches.py} (block-level rebinds):}} \\

\section[Composite]{The composite}

\begin{frame}[fragile]{The composite the walkthrough uses}{One call activates everything}
\begin{exampleblock}{Code (verbatim from \texttt{tutorials/vit\_crp/dinov3\_unfolded/walkthrough.ipynb})}
{\small
\begin{verbatim}
composite = AttnLRPCombinedComposite(

\textbf{Resolves to} (trace through \texttt{AttnLRPCombinedComposite.\_\_init\_\_}):
\begin{itemize}
  \item \alert{layer\_map}: \texttt{nn.Linear} / \texttt{nn.Conv2d} $\to$ \texttt{Epsilon($\eps=10^{-6}$)};
        GELU/LN/Dropout/Identity $\to$ \texttt{Pass} (handled by canonizers).
  \item \alert{TimmViTCanonizer} bundles: \texttt{LayerNormForwardCanonizer}, \texttt{GELUIdentityRuleCanonizer},

  \item \alert{EvaAttentionSubstitutionCanonizer} $+$ \alert{TimmAttentionSubstitutionCanonizer}:

  \item \alert{BilinearMatmulAlphaBetaCanonizer($\alpha=0.5,\beta=0.5$)} on every \texttt{BilinearMatmul}.
  \item \alert{SoftmaxIdentityCanonizer} on every \texttt{SoftmaxAlongLastDim}.
  \item \alert{ScaleByConstantIdentityCanonizer} on every \texttt{ScaleByConstant}.
\end{itemize}

\centering\scriptsize \alert{Net effect}: any supported ViT (24-block Eva DINOv3 or 12-block timm vit\_base) receives the validated AlphaBeta $+$ ratio-residual $+$ uniform-LayerScale recipe — uniformly.
\end{frame}

\section[Concepts]{Concept classes}
% ═══════════════════════════════════════════════════════════════════════════

\begin{frame}{Concepts — the CRP step on top of AttnLRP}{What CRP adds to LRP}
\begin{block}{LRP gives one heat-map per input; CRP gives one heat-map \alert{per concept}.}
A \texttt{Concept} is a \alert{partition of a hidden tensor} into named, addressable
``parts''. To attribute concept $c$ at layer $\ell$, mask the relevance to keep only
parts in $c$, then continue the backward pass as usual.
\end{block}

\begin{itemize}
  \item Each Concept class targets one of the \alert{named submodules from Diagram~B}.
  \item Provides:
    \begin{itemize}
      \item a \texttt{mask\_part(R, part\_id)} method (zero everything not in this part);
      \item a \texttt{reference\_sampling(\ldots)} method for FeatureVisualization.
    \end{itemize}
  \item All six classes share a common base, \texttt{ChannelConcept} (from upstream \texttt{zennit-crp}).
        Per-head attention concepts further share \texttt{\_PerHeadAttentionConcept}.
\end{itemize}

\centering\structure{Six concept classes total}, listed on the next slides with their hook target.
\end{frame}

\begin{frame}{Per-head attention concepts}{Q / K / V / Head — Diagram~B targets}
\centering\small
\begin{tabular}{lll}
\toprule
Concept & Hook & ``Part'' addresses \\
\midrule
\alert{HeadConcept}  & \texttt{blocks.\{i\}.attn.context}   & head $h$, optionally $(h,d)$ \\
\alert{QConcept}     & \texttt{blocks.\{i\}.attn.rope\_q}   & head $h$, optionally $(h,d)$ \\
\alert{KConcept}     & \texttt{blocks.\{i\}.attn.rope\_k}   & head $h$, optionally $(h,d)$ \\
\alert{VConcept}     & \texttt{blocks.\{i\}.attn.v\_id}     & head $h$, optionally $(h,d)$ \\
\bottomrule
\end{tabular}

\vspace{4mm}
\begin{itemize}
  \item All four hook tensors share shape $(B, H, N, d_h)$.
  \item \texttt{dim\_split=False} (default): one part per head — answer \alert{``which head fired?''}.
  \item \texttt{dim\_split=True}: one part per $(h,d)$ channel — finer-grained, answers \alert{``which head\,$\times$\,channel inside that head?''}.
  \item \alert{Spatial-only}: prefix tokens (cls + 4 register) are \alert{excluded} from concept masks
        — those positions go to \texttt{RegisterTokenConcept} (next slide).
  \item All four hook \alert{at the same depth} so per-block side-by-side comparisons (``where in the head's attention does it look?'') are well defined.
\end{itemize}
\end{frame}

\begin{frame}{Token-position concepts}{Output channel + register tokens}
\begin{columns}[T]
\begin{column}{0.5\textwidth}
\begin{exampleblock}{\alert{AttnOutputDimConcept}}
Hook: \texttt{blocks.\{i\}.attn.proj\_drop}, shape $(B, N, D)$.\\
\structure{Part}: channel $d \in [0, D)$ of the post-projection residual.\\
\structure{Tokens covered}: patch tokens only $[5{:}N]$.\\
\structure{Question}: ``Which output channel of attention drove the prediction at this layer?''
\end{exampleblock}
\end{column}
\begin{column}{0.5\textwidth}
\begin{exampleblock}{\alert{RegisterTokenConcept}}
Hook: \texttt{blocks.\{i\}.attn.proj\_drop}, shape $(B, N, D)$.\\
\structure{Part}: token id $t \in [0, 5)$ — the cls token + the 4 DINOv3 register tokens.\\
\structure{Tokens covered}: prefix only $[0{:}5]$.\\
\structure{Question}: ``How much did each register token contribute, and to what?''
\end{exampleblock}
\end{column}
\end{columns}

\vspace{3mm}
\begin{itemize}
  \item \alert{Disjoint by design}: the two concepts hook the same submodule but mask
        complementary token ranges. Together they cover all $N$ tokens, partitioned by role.
  \item Without register-tokens this distinction would not exist — DINOv3 uses register tokens
        as ``signal-sinks'' separate from cls; treating them together would conflate two roles.
\end{itemize}
\end{frame}

\begin{frame}[fragile]{Putting it together}{One block, one HeadConcept attribution}
\textbf{What happens when the user runs:}
{\scriptsize
\begin{semiverbatim}
res = attribution(x, [{"y":[class_id], "head_concept":[(layer, head)]}], composite)
\end{semiverbatim}
}

\begin{enumerate}
  \item Composite \texttt{context} enters $\Rightarrow$ all canonizers \texttt{apply}.\\
        Eva attention is now unfolded; every named submodule has its rule wired in.
  \item Forward pass through the model with the user's input.
  \item \texttt{HeadConcept.mask\_part(R\_pre\_attn, head)} zeros every head other than the requested one
        in the relevance tensor flowing back from the logit.
  \item Backward pass routes the masked relevance through:
        \begin{itemize}
          \item \texttt{context} (bilinear AlphaBeta) $\to$ \texttt{softmax} (identity) $\to$ \texttt{qk\_scores} (bilinear AlphaBeta)
          \item Through \texttt{rope\_q/k} and \texttt{q\_norm/k\_norm} ($\eps$-LRP)
          \item Through residuals (ratio split) and LayerScale (uniform $\half$ allocation).
        \end{itemize}
  \item Result is per-input-pixel relevance attributable to \alert{exactly this head} at \alert{exactly this block}.
  \item Composite \texttt{context} exits $\Rightarrow$ all canonizers \texttt{remove}; model is back to vanilla.
\end{enumerate}
\end{frame}

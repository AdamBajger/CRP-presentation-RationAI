\section[Motivation]{Motivation}
% ═══════════════════════════════════════════════════════════════════════════

\begin{frame}{Motivation}
\begin{itemize}
  \item ViTs win vision benchmarks. We want \alert{post-hoc explanations} that are:
    \begin{itemize}
      \item \structure{faithful} (conserve the prediction),
      \item \structure{spatial} (pixel-level heat-maps),
      \item \structure{conceptual} (per-head, per-channel, per-token).
    \end{itemize}
  \item Gradient saliency loses conservation through softmax, bilinear matmul,
        residual adds, LayerScale.
  \item \alert{AttnLRP}~\cite{achtibat2024attnlrp} — transformer-aware LRP rules.
  \item \alert{CRP}~\cite{achtibat2023crp} — lifts attribution to per-concept maps.
  \item Our work: ground both into a single \texttt{zennit}-compatible composite for
        DINOv3~\cite{darcet2024dinov3} and standard timm ViTs~\cite{dosovitskiy2021vit}.
\end{itemize}
\end{frame}

% ═══════════════════════════════════════════════════════════════════════════
\section[LRP]{LRP}
% ═══════════════════════════════════════════════════════════════════════════

\begin{frame}{LRP-0 and $\eps$-LRP}
\begin{block}{LRP-0 (Bach et al.\ 2015)}
For $y_j = \sum_i w_{ij}\, x_i + b_j$:
\[
  \Rel_i = \sum_j \frac{w_{ij}\, x_i}{\sum_k w_{kj}\, x_k}\, \Rel_j
\]
Exact conservation on linear layers.
\end{block}
\begin{block}{$\eps$-LRP (stabilised)}
\[
  \Rel_i = \sum_j \frac{w_{ij}\, x_i}{\eps\cdot \mathrm{sign}(z_j) + z_j}\, \Rel_j
\]
$\eps$ kills small-denominator amplification. Conservation $\sum \Rel_i \approx \sum \Rel_j$ within $\eps$.
\end{block}

\structure{Problem}: softmax, bilinear matmul, residual adds, LayerScale are NOT linear.
Stock LRP cannot route relevance through them.
\end{frame}

\begin{frame}{AttnLRP}
Per-non-linearity rules that preserve LRP-0 conservation~\cite{achtibat2024attnlrp}:

\begin{itemize}
  \item \alert{Identity rule} for normalisations (Eq.~9) — softmax, LayerNorm, GELU.
  \item \alert{Bilinear rule} (Prop.~3.3) for $Q@K^\top$ and $\mathrm{attn}@V$.
  \item \alert{Stop-gradient on std} inside LayerNorm.
\end{itemize}

\begin{block}{Our additions}
\begin{itemize}
  \item \alert{AlphaBeta on bilinear} — Bach~\cite{bach2015lrp} generalised; better magnitude control.
  \item \alert{Residual ratio} (Otsuki split) and \alert{LayerScale uniform rule}.
  \item \alert{Attention unfolding}: substitute attention with named-submodule form
        so concepts can hook on $Q$, $K$, $V$, head, channel, token positions.
\end{itemize}
\end{block}
\end{frame}

\section[Rules]{LRP rules}
% ═══════════════════════════════════════════════════════════════════════════

\begin{frame}{Rules}
\small
\begin{tabular}{lll}
\toprule
Module & Rule & Source \\
\midrule
\texttt{Linear}, \texttt{Conv2d}      & $\eps$-LRP                          & \cite{bach2015lrp}; \texttt{zennit.Epsilon}~\cite{anders2021zennit} \\
\texttt{LayerNorm}                    & Stop-grad on std + identity         & AttnLRP \S 3.2.2~\cite{achtibat2024attnlrp} \\
\texttt{GELU}                         & Identity (Eq.~9)                    & \cite{achtibat2024attnlrp} \\
\texttt{softmax}                      & Identity                            & \cite{achtibat2024attnlrp} \\
\texttt{scale\_q} (constant mul)      & Identity                            & graph constant \\
\texttt{qk\_scores}, \texttt{context} & \alert{AlphaBeta on bilinear}       & \cite{bach2015lrp} + our derivation \\
Residual additions                    & \alert{Otsuki ratio split}          & magnitude-weighted \\
LayerScale                            & \alert{Uniform $\half$ allocation}  & AttnLRP Eq.~7~\cite{achtibat2024attnlrp} \\
\bottomrule
\end{tabular}

\vspace{4mm}
\structure{Vanilla forward, LRP only on backward.}
Forwards are bit-identical; only the autograd graph changes during attribution.
\end{frame}

\begin{frame}{$\eps$-LRP}
For $z_j = \sum_i w_{ij}\, x_i + b_j$ in any \texttt{Linear}:
\[
  \Rel_i = \sum_j \frac{w_{ij}\, x_i}{\eps\cdot\mathrm{sign}(z_j) + z_j}\, \Rel_j
\]

\begin{itemize}
  \item Workhorse — every \texttt{Linear} / \texttt{Conv2d} (\texttt{qkv}, \texttt{proj}, \texttt{fc1}, \texttt{fc2}, \ldots).
  \item Conservation exact modulo $\eps$.
\end{itemize}
\end{frame}

\begin{frame}{Identity rule}
For an element-wise nonlinearity $y_i = f(x_i)$, LRP-0 reduces to
\[
  \Rel_i \;=\; \frac{y_i}{\eps\cdot\mathrm{sign}(y_i) + y_i}\, \Rel^{\mathrm{out}}_i
  \;\approx\; \Rel^{\mathrm{out}}_i \qquad (|y_i| \gg \eps).
\]

\begin{itemize}
  \item \texttt{softmax}: relevance passes \alert{straight through} — does NOT route via Jacobian.
  \item \texttt{LayerNorm}: \texttt{stop\_gradient}(std) $\to$ identity on affine.
  \item \texttt{GELU}, \texttt{scale\_q} (constant multiplier): same form.
\end{itemize}

Conservation: $\sum \Rel_i = \sum \Rel^{\mathrm{out}}_i$.
\end{frame}

\begin{frame}{Bilinear rule: AlphaBeta}
For $Y = A\,B$, decompose by sign: $A^\pm = \max/\min(A, 0)$, $B^\pm = \max/\min(B, 0)$.
\[
  Y^+ = A^+B^+ + A^-B^- \geq 0, \qquad
  Y^- = A^+B^- + A^-B^+ \leq 0
\]
Define $F = \Rel_Y/(Y^+ + \eps)$, $G = \Rel_Y/(Y^- - \eps)$. Then
\[
  \Rel_A = \half \Bigl\{ A^+ \odot \bigl[\alpha\,(FB^{+\top}) + \beta\,(GB^{-\top})\bigr]
                       + A^- \odot \bigl[\alpha\,(FB^{-\top}) + \beta\,(GB^{+\top})\bigr] \Bigr\}
\]
(symmetric for $\Rel_B$).

\begin{itemize}
  \item Conservation: $\sum \Rel_A + \sum \Rel_B = (\alpha+\beta)\sum \Rel_Y$; exact when $\alpha+\beta=1$.
  \item \alert{Default} $\alpha=0.5, \beta=0.5$ — balanced, tightest magnitude control.
  \item AttnLRP's $2Y+\eps$ rule amplifies by $\sim 1/\eps$ at near-orthogonal $Q,K$.
        Sign-splitting denominators avoid this structurally.
\end{itemize}
\end{frame}

\begin{frame}{Residual rule: Otsuki ratio split}
For $y = x + \mathrm{branch}$:
\[
  \Rel_x = \Rel_y \odot \frac{|x|}{|x| + |\mathrm{branch}| + \eps},
  \qquad
  \Rel_{\mathrm{branch}} = \Rel_y \odot \frac{|\mathrm{branch}|}{|x| + |\mathrm{branch}| + \eps}
\]

\begin{itemize}
  \item Magnitude-weighted: bigger-signal branch gets more relevance.
  \item Bare autograd: $\Rel_x = \Rel_{\mathrm{branch}} = \Rel_y$ — sums to $2\Rel_y$
        per block $\Rightarrow$ $2^{24}\!\times$ blow-up on a 24-block ViT-L.
  \item Conservation: $\sum \Rel_x + \sum \Rel_{\mathrm{branch}} \approx \sum \Rel_y$.
\end{itemize}
\end{frame}

\begin{frame}{LayerScale: uniform $\half$ allocation}
LayerScale~\cite{touvron2021cait}: $\mathrm{branch} \to \gamma \odot \mathrm{branch}$ with $\gamma \in \reals^D$ init $\sim 10^{-4}$.

\begin{block}{Why bare autograd fails}
$\nabla_{\mathrm{branch}} = \nabla_y \cdot \gamma$ with $\gamma \sim 10^{-4}$, 24 blocks $\times$ 2 LayerScales
$\Rightarrow$ $\Rel$ deflates by $10^{-200}$ — numerical death.
\end{block}

\begin{block}{Fix (AttnLRP Eq.~7)}
Treat $\gamma$ as a leaf parameter absorbing half the relevance:
\[
  \Rel_{\mathrm{branch}} = \half \Rel_y,\qquad \Rel_\gamma = \half \Rel_y\ \mbox{(absorbed)}.
\]
Replaces ``$\times\gamma$'' with constant ``$\times\half$'' per LayerScale.
\end{block}
\end{frame}

\begin{frame}{Conservation}
\centering\small
\begin{tabular}{lll}
\toprule
Rule & Conservation & Comment \\
\midrule
$\eps$-LRP on Linear     & $\sum \Rel_i = \sum \Rel_j$ mod $\eps$       & exact \\
Identity (sm / LN / GELU) & $\sum \Rel_i = \sum \Rel_j$ exact            & passes through \\
Bilinear AlphaBeta       & $= (\alpha+\beta)\sum \Rel_Y$               & exact when $\alpha+\beta=1$ \\
Residual ratio           & $\sum \Rel_x + \sum \Rel_b \approx \sum \Rel_y$ & magnitude-weighted \\
LayerScale uniform       & $\Rel_{\mathrm{branch}} = \half \Rel_y$       & $\gamma$ absorbs the rest \\
\bottomrule
\end{tabular}

\vspace{4mm}
\begin{itemize}
  \item Bare autograd on 24-block Eva DINOv3 inflates $\sum\!\Rel$ by $\sim\!10^{21}$.
  \item Composite-corrected: $\sum\!\Rel$ within $\sim\!10^0$–$10^1$ of target logit.
  \item Audit metric: $\sum\!\Rel\,/\,\mathrm{logit}$ per block.
\end{itemize}
\end{frame}

\section[Canonizers]{Canonizers}
% ═══════════════════════════════════════════════════════════════════════════

\begin{frame}{Canonizers}
Reversible model transformations applied inside the composite \texttt{context}:

\begin{itemize}
  \item \alert{Forward rebind} — replace the module's \texttt{forward}.
        Forward output unchanged; backward routes through the rule's
        autograd \texttt{Function}.
  \item \alert{Substitution} — swap monolithic attention for an unfolded
        version with named submodules (\texttt{qk\_scores}, \texttt{softmax},
        \texttt{context}, \texttt{*\_lrp\_probe}, \ldots).\\
        Required because per-tensor rules cannot attach to a single
        monolithic \texttt{Attention.forward}.
\end{itemize}

\centering\structure{Both types are reversible.}
\end{frame}

\section[Composite]{Composite}
% ═══════════════════════════════════════════════════════════════════════════

\begin{frame}{Composite}
\centering\large
\texttt{AttnLRPCombinedComposite($\alpha{=}0.5,\beta{=}0.5$, residual\_lrp='ratio', layerscale\_uniform=True)}

\vspace{6mm}
\begin{itemize}
  \item Linear / Conv $\to$ $\eps$-LRP
  \item LayerNorm, GELU, softmax, scale $\to$ Identity rule
  \item All bilinears $\to$ AlphaBeta($0.5, 0.5$)
  \item Residual adds $\to$ Otsuki ratio
  \item LayerScale $\to$ uniform $\half$
  \item Attention $\to$ unfolded substitution (timm + Eva, both always-on)
\end{itemize}

\centering\structure{One call activates everything.} Any supported ViT (Eva DINOv3, timm \texttt{vit\_base}, our \texttt{vit\_small})
receives the same validated recipe.
\end{frame}

% ═══════════════════════════════════════════════════════════════════════════
\section[Concepts]{Concepts}
% ═══════════════════════════════════════════════════════════════════════════

\begin{frame}{Concepts}
\begin{block}{LRP $\to$ one heatmap per input. CRP $\to$ one heatmap per concept.}
A \alert{concept} partitions a hidden tensor into named ``parts''. To
attribute concept $c$ at layer $\ell$: mask all parts $\neq c$, continue
backward as usual.
\end{block}

\begin{block}{Three concepts, four hookable sites — all share one shape}
Every site is $(B, N, D)$ — the 3D tensor at the attention boundary.
Any concept hooks at any site; meaning shifts with the site.
\end{block}

\begin{itemize}
  \item \alert{HeadConcept} — partitions $D$ into $H$ contiguous head slices ($D/H$ wide).
  \item \alert{EmbeddingDimConcept} — one part per single embedding dim ($D$ parts total).
  \item \alert{TokenConcept} — one part per token position; optional
        \texttt{token\_filter=slice(...)} restricts to cls / register / spatial.
\end{itemize}
\end{frame}

\begin{frame}{HeadConcept}
For relevance $\Rel \in \reals^{B \times N \times D}$ at any 3D site,
$\Rel$ reshaped to $(B, N, H, d_h)$ has heads on a separate axis.

\begin{block}{Mask for head $h$}
\[
  \Rel'_{b, n, h', d} = \begin{cases} \Rel_{b, n, h', d} & h' = h \\ 0 & \mbox{otherwise} \end{cases}
\]
\end{block}

\begin{block}{Per-head attribution}
\[
  a_{b, h} = \sum_{n \in \mathit{token\_filter}}\;\; \sum_{d=0}^{d_h-1} \Rel_{b, n, h, d}
\]
\end{block}

\structure{Question}: ``which head fired on which input regions?''
\end{frame}

\begin{frame}{EmbeddingDimConcept}
Finer than HeadConcept: one part per single dim of $D$.

\begin{block}{Mask for dim set $S \subseteq \{0,\ldots,D-1\}$}
\[
  \Rel'_{b, n, d} = \begin{cases} \Rel_{b, n, d} & d \in S \\ 0 & \mbox{otherwise} \end{cases}
\]
\end{block}

\begin{block}{Per-dim attribution}
\[
  a_{b, d} = \sum_{n \in \mathit{token\_filter}} \Rel_{b, n, d}
\]
\end{block}

\structure{Question}: ``which single embedding direction carries the prediction at this layer?''
\end{frame}

\begin{frame}{TokenConcept}
Partitions on the token axis. For prefix-vs-spatial separation, use
\texttt{token\_filter} (e.g.\ \texttt{slice(0, 5)} = cls + 4 register tokens
on DINOv3; \texttt{slice(5, None)} = spatial patches only).

\begin{block}{Mask for token set $T \subseteq [0, N)$}
\[
  \Rel'_{b, n, d} = \begin{cases} \Rel_{b, n, d} & n \in T \\ 0 & \mbox{otherwise} \end{cases}
\]
\end{block}

\begin{block}{Per-position attribution}
\[
  a_{b, n} = \sum_{d=0}^{D-1} \Rel_{b, n, d}, \qquad n \in \mathit{token\_filter}
\]
\end{block}

\structure{Question}: ``did this cls / register / patch position drive the prediction?''\\
DINOv3 register tokens~\cite{darcet2024dinov3}: 4 ``signal-sinks'' separate from cls.
\end{frame}

\begin{frame}{One attribution trace}
\textbf{User runs}\\[1mm]
\centering\small\texttt{attribution(x, [\{"y":[c], $\ell$:[$h$]\}], composite, mask\_map=HeadConcept.mask)}\\[3mm]

\raggedright
\begin{enumerate}
  \item Composite \texttt{context} enters $\Rightarrow$ canonizers apply,
        attention is unfolded, every named submodule has its rule wired.
  \item Forward pass through the model on input $x$.
  \item \texttt{HeadConcept.mask} zeros every head $\neq h$ in the
        relevance flowing back from logit $y$.
  \item Backward through:
        \begin{itemize}
          \item \texttt{context} (AlphaBeta) $\to$ \texttt{softmax} (identity) $\to$ \texttt{qk\_scores} (AlphaBeta)
          \item \texttt{rope\_q/k}, \texttt{q\_norm/k\_norm} ($\eps$-LRP)
          \item Residuals (ratio split), LayerScale (uniform $\half$).
        \end{itemize}
  \item Result: per-pixel relevance for \alert{exactly this head} at \alert{exactly this block}.
  \item \texttt{context} exits $\Rightarrow$ canonizers remove, model back to vanilla.
\end{enumerate}
\end{frame}

\section[Findings]{Key findings}
% ═══════════════════════════════════════════════════════════════════════════

\begin{frame}{Key findings so far}
\begin{itemize}
  \item \alert{Bare AttnLRP fails on DINOv3.} 24 EvaBlocks with LayerScale
        $\sim\!10^{-4}$ + the $2Y+\eps$ bilinear rule overflow fp32 by
        mid-depth. No single literature remedy fixes it; the working
        composite needs \alert{three} rules simultaneously: AlphaBeta
        bilinear, Otsuki ratio residual, uniform LayerScale.
  \item \alert{AlphaBeta-on-bilinear} replaces $2Y+\eps$. Sign-splitting
        denominators bound per-element amplification structurally —
        \alert{$\sim$ 19 OOM} smaller $\max|R|$ vs the standard rule on
        DINOv3 ViT-L, while preserving spatial focus.
  \item \alert{Unfolded attention enables CRP.} Stock attention has no
        named submodules. Substituting in a forward-equivalent unfolded
        form exposes $Q$, $K$, $V$, scores, softmax, output as concept hooks.
  \item \alert{Single shape contract.} All four 3D hookable sites carry
        $(B, N, D)$; any of the three concepts attaches at any site.
        Three classes replaced our earlier six-class taxonomy.
  \item \alert{Residual flow on FunnyBirds vit\_base}: 70--82\% of
        relevance flows through the skip path mid-stack — motivates the
        move to smaller (12-block) backbones.
\end{itemize}
\end{frame}

\begin{frame}{Trained \texttt{vit\_small} on three datasets}
\centering\small
\begin{tabular}{lllp{0.32\textwidth}}
\toprule
Dataset & train-val acc & test acc & Interpretation \\
\midrule
\alert{dSprites}      & $0.999$ & $0.999$ & 3-class shape task — trivial; ceiling. \\
\alert{FunnyBirds}    & (in progress) & — & 50-class part-based; ImageNet recipe (RandAug + Mixup + label smoothing + cosine + LLRD). \\
\alert{ColoredMNIST}  & $0.978$ & $0.127$ & \alert{intentionally biased}: model learned colour shortcut. Train-val high, test near chance. \\
\bottomrule
\end{tabular}

\vspace{4mm}
\begin{itemize}
  \item ColoredMNIST is the deliverable \alert{bias probe} — we know the
        model's failure mode (colour, not shape). XAI heatmaps should
        light up colour-bearing pixels, not stroke geometry. If they
        don't, the method is mis-identifying the cause of the prediction.
  \item The other two are sanity-checks: known parts (FunnyBirds), known
        shape pixels (dSprites). Heatmaps should align with the ground
        truth.
\end{itemize}
\end{frame}

\begin{frame}{Cross-check XAI baselines}
We ship the trained \texttt{vit\_small} with two published baseline methods
for manual sanity checking before applying CRP:

\begin{itemize}
  \item \alert{LeGrad}~\cite{bousselham2024legrad} — gradient of the
        prediction w.r.t.\ attention weights; aggregated per block.
  \item \alert{Chefer's rollout}~\cite{chefer2021transformer} — per-block
        $(\mathrm{grad} \odot A)_+$ averaged over heads, rolled up across
        blocks via $R \leftarrow R + R_\mathrm{block} R$.
\end{itemize}

\centering\structure{Both hook the post-softmax attention weight tensor.}\\
No composite needed — they are stock-model methods.
The walkthrough notebook runs them next to our AttnLRP / CRP attributions on the same focal image,
so the user can compare the three approaches visually.
\end{frame}
